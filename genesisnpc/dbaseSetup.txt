CREATE DATABASE genesisNPC;

CREATE TABLE genesisUser (
user VARCHAR(50) NOT NULL, 
userpwd VARCHAR(255) NOT NULL,
security VARCHAR(255) NOT NULL,
userId BIGINT(255) NOT NULL AUTO_INCREMENT PRIMARY KEY,
email VARCHAR(255) NOT NULL,
hash VARCHAR(255) NOT NULL,
active INT(1) NOT NULL DEFAULT '0',
UNIQUE (user)
);

CREATE TABLE DataList (
UserID BIGINT(255) NOT NULL, 
DateCreated VARCHAR(255) NOT NULL,
type VARCHAR(2) NOT NULL,
name VARCHAR(255) NOT NULL
);

CREATE TABLE genesisGuest (
guestAmount BIGINT(255) NOT NULL AUTO_INCREMENT PRIMARY KEY, 
type VARCHAR(255) NOT NULL,
guestLogin BIGINT(255) NOT NULL,
name VARCHAR(255) NOT NULL
);
